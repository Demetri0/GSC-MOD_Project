	function frpgCalc(){
		scoreTeam1 = 0;
		scoreTeam2 = 0;
		if(document.all["t1-health"].length){
			for(i=0;i < document.all["t1-health"].length;i++){
				scoreTeam1 += parseInt(document.all["t1-health"][i].value);
				scoreTeam1 += parseInt(document.all["t1-defence"][i].value)/1.2;
				scoreTeam1 += parseInt(document.all["t1-strong"][i].value)/0.9;
				scoreTeam2 += parseInt(document.all["t2-health"][i].value);
				scoreTeam2 += parseInt(document.all["t2-defence"][i].value)/1.2;
				scoreTeam2 += parseInt(document.all["t2-strong"][i].value)/0.9;
			}
		}else{
			scoreTeam1 += parseInt(document.all["t1-health"].value);
			scoreTeam1 += parseInt(document.all["t1-defence"].value)/1.2;
			scoreTeam1 += parseInt(document.all["t1-strong"].value)/0.9;
			scoreTeam2 += parseInt(document.all["t2-health"].value);
			scoreTeam2 += parseInt(document.all["t2-defence"].value)/1.2;
			scoreTeam2 += parseInt(document.all["t2-strong"].value)/0.9;
		}
		luck();
		scoreTeam2 += bonus;
		scoreTeam2 = Math.round(scoreTeam2);
		scoreTeam1 = Math.round(scoreTeam1);
		//alert(scoreTeam1+'<1!==!2>'+scoreTeam2);//Вывод кол-ва очков
		if(isNaN(scoreTeam1) || isNaN(scoreTeam2)){alert('Неверно заполнены поля!')}else{
			if(scoreTeam1>scoreTeam2){alert('Побеждает '+document.all["team1"].value);}
			if(scoreTeam2>scoreTeam1){alert('Побеждает '+document.all["team2"].value);}
			if(scoreTeam1==scoreTeam2){alert('Ничья!');}
		}
	}
